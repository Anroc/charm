[docs]    def P(self, coeff, x):
        share = 0
        # evaluate polynomial
        for i in range(0, len(coeff)):
            share += (coeff[i] * (x ** i))
        return share



[docs]    def genShares(self, secret, k=0, n=0, q=None, x_points=None):
        if(k <= n):
            if q == None: 
                q = [self.elem.random(ZR) for i in range(0, k)]
                q[0] = secret

            if x_points == None: # just go from 0 to n
                shares = [self.P(q, i) for i in range(0, n+1)] # evaluating poly. q at i for all i
            else:
                shares = {}
                for i in range(len(x_points)):
                    shares[i] = (x_points[i], self.P(q, x_points[i]))

        if self.verbose:
            print('Secret: %s' % secret)

                print("a %s: %s" % (i, q[i]))

            if x_points == None:

                    print('Share %s: %s' % (i, shares[i]))
            else:
                for i in range(len(x_points)):
                    print('Share %s: %s' % (i, shares[i]))
            
        return shares

[docs]    def recoverCoefficients(self, list):
        coeff = {}
        for i in list:
            result = 1
            for j in list:
                if not (i == j):
                    # lagrange basis poly
                    result *= (0 - j) / (i - j)
            if self.verbose: print("coeff '%d' => '%s'" % (i, result))
            coeff[i] = result
        return coeff

[docs]    def recoverSecret(self, shares):
        list = shares.keys()
        if self.verbose: print(list)
        coeff = self.recoverCoefficients(list)
        if self.verbose: print("coefficients: ", coeff)
        secret = 0
        for i in list:
            secret += (coeff[i] * shares[i])

        return secret
