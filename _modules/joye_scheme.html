        prod=pai.encode(pk['n2'],x1*x2)
        inter=pai.encode(pk['n2'],integer(self.r)**integer(integer(-1)*k0))
        inter2=(integer(prod)/integer(inter))
        sumres=integer(inter2%pk['n2']-integer(1)%pk['n2'])/pk['n']
        return sumres
