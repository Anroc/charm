        if(p == 0):
            pprime = randomPrime(secparam)

                pprime = randomPrime(secparam)

        N = p * q

        e = randomPrime(le)

        ls = ln + lm + l
        s = integer(randomBits(ls))

        phi_N = (sk['p']-1)*(sk['q']-1)
        e2 = e % phi_N
    
        v = (((pk['a'] ** m)*(pk['b'] ** s)*pk['c']) ** (e2 ** -1)) % pk['N']

        sig = { 'e':e, 's':s, 'v':v }

        return sig

        if debug: print("\nVERIFY\n\n")

        lhs = (sig['v'] ** sig['e']) % pk['N']
        rhs = ((pk['a'] ** m)*(pk['b'] ** sig['s'])*pk['c']) % pk['N']
        
        if (sig['e'] <= 2**(le - 1) or sig['e'] >= 2**(le)):
            return False

        if(lhs == rhs):
            return True

        return False
